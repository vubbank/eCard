 79     /**
 80      * Concats provided array into readeable html attributes string.
 81      *
 82      * @example From array of ['style'=>'color:white', 'title'=>'test'] creates resulting string
 83      * style="color:white" title="test"
 84      *
 85      * @param  Array $htmlAttributes  Array containing HTML attribute and value. [attribute=>value]
 86      * @return string                 Concated string of html attributes
 87      */
 88     public static function concatHtmlAttributes($htmlAttributes = [])
 89     {
 90       if (!is_array($htmlAttributes)) {
 91         throw new VubException('HtmlAttributes have to be associative array [atttributeName => attributeValue]', 500);
 92       }
 93 
 94       $htmlAttributesString = [];
 95       foreach ($htmlAttributes as $attributeName => $attributeValue) {
 96 
 97         $htmlAttributesString[] = $attributeName . '="' . $attributeValue . '"';
 98       }
 99 
100       return implode(' ', $htmlAttributesString);
101     }
